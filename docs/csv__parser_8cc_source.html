161 merge_csv_vcd_table_.clear();

163
164 /* Determine whether it is a vector signal*/
165 std::string label, width;
166 if ((iter.first.length() - iter.first.rfind(']')) == 1 &&
167 (iter.first.rfind('[') != std::string::npos) &&
168 iter.first.length() >= 4) {
169 bool read_label = true;
170 for (auto &pos : iter.first) {
171 if (pos == '[')
172 read_label = false;
173 if (read_label)
174 label += pos;
175 }
176
177 bool read_width = false;
178 for (auto &pos : iter.first) {
179 if (pos == '[')
180 read_width = true;
181 if (read_width)
182 width += pos;
183 }
184
185 } else
186 label = iter.first;
187
188 if (csv_vcd_signal_table_.find(label) == csv_vcd_signal_table_.end())
189 continue;
190
191 /* Determine what the starting bit width is.*/
192 if (csv_vcd_signal_table_.find(label).value().declare_width_start != 0) {
193 int width_pos = std::stoi(width.substr(1, width.find(']') - 1));
194 width_pos = width_pos - csv_vcd_signal_table_.find(label).value().declare_width_start;
195 width.clear();
196 width = '[' + std::to_string(width_pos) + ']';
197 }
198 std::string signal_label;
199 if (width.empty())
200 signal_label = csv_vcd_signal_table_.find(label).value().vcd_signal_label;
201 else
202 signal_label = csv_vcd_signal_table_.find(label).value().vcd_signal_label + width;
203 merge_csv_vcd_table_.insert(std::pair<std::string, struct CSVSignalStatisticStruct>(signal_label, iter.second));
204 }
